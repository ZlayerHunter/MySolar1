package ru.slrhntr.solarview.control;

import com.badlogic.gdx.math.MathUtils;
import com.badlogic.gdx.math.Polygon;

public class VectorMercury {
    private Polygon mercuryBounds;
    public VectorMercury(Polygon mercuryBounds){
        this.mercuryBounds = mercuryBounds;

    }
    public void handle(){
        float speed = 100f;
        //earthBounds.setPosition(earthBounds.getX(),earthBounds.getY());
        mercuryBounds.rotate(1.2f);
        mercuryBounds.setPosition(mercuryBounds.getX() + MathUtils.cosDeg(mercuryBounds.getRotation()),
                mercuryBounds.getY() + MathUtils.sinDeg(mercuryBounds.getRotation()));

    }

}
